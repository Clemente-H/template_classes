\section{Introducción}

\begin{frame}{El problema del conocimiento congelado}
\begin{itemize}
    \item Los LLMs tienen conocimiento estático
    \item No acceden a datos privados
    \item Pueden alucinar
\end{itemize}
\end{frame}

\begin{frame}{Motivación}
\begin{block}{Pregunta clave}
¿Cómo damos acceso a información actualizada sin reentrenar?
\end{block}
\end{frame}
